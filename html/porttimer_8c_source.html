39 /**
40  * @brief Enables a match interrupt that fires when the terminal count
41  * matches the match counter value.
42  * @param pTMR : Pointer to timer IP register address
43  * @param matchnum : Match timer, 0 to 3

62 /**

64  * @param pTMR : Pointer to timer IP register address

89 /* Resets the timer terminal and prescale counts to 0 */
90 void Chip_TIMER_Reset(LPC_TIM_TypeDef *pTMR)
91 {
92  uint32_t reg;
93 
94  /* Disable timer, set terminal count to non-0 */
95  reg = pTMR->TCR;
96  pTMR->TCR = 0;
97  pTMR->TC = 1;
98 
99  /* Reset timer counter */
100  pTMR->TCR = TIMER_RESET;
101 
102  /* Wait for terminal count to clear */
103  while (pTMR->TC != 0) {}
104 
105  /* Restore timer state */
106  pTMR->TCR = reg;
107 }

109  * @brief For the specific match counter, enables reset of the terminal count register when a match occurs

113  */
