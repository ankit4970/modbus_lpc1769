83 BOOL xMBPortSerialInit( UCHAR ucPORT, ULONG ulBaudRate, UCHAR ucDataBits, eMBParity eParity )
84 {
85  //simISR.attach_us(&prvvUARTISR,1000); // Cam - attach prvvUARTISR to a 1mS ticker to simulate serial interrupt behaviour
86  // 1mS is just short of a character time at 9600 bps, so quick enough to pick
87  // up status on a character by character basis.
88 
89  //const uint32_t Baud_Rate = 9600;
90 
91  // Step 1: UART2 power/clock control bit(24) of PCONP
92  LPC_SC->PCONP &=~ (1<<24);
93  LPC_SC->PCONP |= (1<<24);
94 
95  // Step 2: PCLKSEL1 bit 16 and 17 of Peripheral Clock Selection register1 for Uart2, controls the rate of the clock signal
96  // that will be supplied to the corresponding peripheral Function peripheral clock selection
97  // 00 CCLK/4
98  // 01 CCLK
99  // 10 CCLK/4
100  // 11 CLK/4
101  LPC_SC->PCLKSEL1 &=~(3<<16);
102  LPC_SC->PCLKSEL1 |= (1<<16);
103 
104  // Step 3:// table 280
105  LPC_UART2->LCR &=~ (3<<0 | 1<<7);
106 
107  // LCR bit o and 1, Word length select
108  LPC_UART2->LCR |= (3<<0); //00 5-bit character length
109  //01 6-bit character length
110  //10 7-bit character length
111  //11 8-bit character length
112 
113  //Enable, Divisor latch access bit(DLAB)
114  LPC_UART2->LCR |= (1<<7);
115  LPC_UART2->DLM = 0x00;
116  //LPC_UART3->DLL =0x38;
117  LPC_UART2->DLL =( sys_get_cpu_clock() / (16 * ulBaudRate));
118 
119  // Step 4: FIFO not use:
120  LPC_PINCON->PINSEL0 &=~ (3<<20 | 3<<22);
121  LPC_PINCON->PINSEL0 |= (1<<20 | 1<<22); //TXD3 | RXD3
122  LPC_PINCON->PINMODE4 &=~ (3<<20 | 3<<22);
123  LPC_PINCON->PINMODE4 |= (3<<22); // 11 pull down
124 
125  // Step 6: Disable, Divisor latch access bit(DLAB)
126  LPC_UART2->LCR &=~ (1<<7);
127  NVIC_EnableIRQ(UART2_IRQn);
128 
129  //enable receive data available interrupt for uart2
130  LPC_UART2->IER &=~ (1<<0);
131  LPC_UART2->IER |= (1<<0);
132 
133  return TRUE;
134 }
