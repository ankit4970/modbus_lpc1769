80 eMBErrorCode eMBRTUInit( UCHAR ucSlaveAddress, UCHAR ucPort, ULONG ulBaudRate, eMBParity eParity )
81 {
82  eMBErrorCode eStatus = MB_ENOERR;
83  ULONG usTimerT35_50us;
84 
85  ( void )ucSlaveAddress;
86  ENTER_CRITICAL_SECTION( );
87 
88  /* Modbus RTU uses 8 Databits. */
89  if( xMBPortSerialInit( ucPort, ulBaudRate, 8, eParity ) != TRUE )
90  {
91  eStatus = MB_EPORTERR;
92  }
93  else
94  {
95  /* If baudrate > 19200 then we should use the fixed timer values
96  * t35 = 1750us. Otherwise t35 must be 3.5 times the character time.
97  */
98  if( ulBaudRate > 19200 )
99  {
100  usTimerT35_50us = 35; /* 1800us. */
101  }
102  else
103  {
104  /* The timer reload value for a character is given by:
105  *
106  * ChTimeValue = Ticks_per_1s / ( Baudrate / 11 )
107  * = 11 * Ticks_per_1s / Baudrate
108  * = 220000 / Baudrate
109  * The reload for t3.5 is 1.5 times this value and similary
110  * for t3.5.
111  */
112  usTimerT35_50us = ( 7UL * 220000UL ) / ( 2UL * ulBaudRate );
113  }
114  if( xMBPortTimersInit( ( USHORT ) usTimerT35_50us ) != TRUE )
115  {
116  eStatus = MB_EPORTERR;
117  }
118  }
119 
120  EXIT_CRITICAL_SECTION( );
121 
122  return eStatus;
123 }

125 void eMBRTUStart( void )
126 {
127  ENTER_CRITICAL_SECTION();
128 
129  /* Initially the receiver is in the state STATE_RX_INIT. we start
130  * the timer and if no character is received within t3.5 we change
131  * to STATE_RX_IDLE. This makes sure that we delay startup of the
132  * modbus protocol stack until the bus is free.
133  */
134  eRcvState = STATE_RX_INIT;
135  vMBPortSerialEnable(TRUE, FALSE);
136  vMBPortTimersEnable();
137 
138  EXIT_CRITICAL_SECTION( );
139 }

149 eMBErrorCode eMBRTUReceive( UCHAR * pucRcvAddress, UCHAR ** pucFrame, USHORT * pusLength )
150 {
151  BOOL xFrameReceived = FALSE;
152  eMBErrorCode eStatus = MB_ENOERR;
153 
154  ENTER_CRITICAL_SECTION( );
155  assert( usRcvBufferPos < MB_SER_PDU_SIZE_MAX );
156 
157  /* Length and CRC check */
158  if( ( usRcvBufferPos >= MB_SER_PDU_SIZE_MIN )
159  && ( usMBCRC16( ( UCHAR * ) ucRTUBuf, usRcvBufferPos ) == 0 ) )
160  {
161  /* Save the address field. All frames are passed to the upper layer
162  * and the decision if a frame is used is done there.
163  */
164  *pucRcvAddress = ucRTUBuf[MB_SER_PDU_ADDR_OFF];
165 
166  /* Total length of Modbus-PDU is Modbus-Serial-Line-PDU minus
167  * size of address field and CRC checksum.
168  */
169  *pusLength = ( USHORT )( usRcvBufferPos - MB_SER_PDU_PDU_OFF - MB_SER_PDU_SIZE_CRC );
170 
171  /* Return the start of the Modbus PDU to the caller. */
172  *pucFrame = ( UCHAR * ) & ucRTUBuf[MB_SER_PDU_PDU_OFF];
173  xFrameReceived = TRUE;
174 
175  // Added by Cam
176  // Now that the poll routine knows about the received frame,
177  // clear the receive buffer position ready for the next frame received
178  usRcvBufferPos = 0;
179 
180  }
181  else
182  {
183  eStatus = MB_EIO;
184  }
185 
186  EXIT_CRITICAL_SECTION( );
187  return eStatus;
188 }

190 eMBErrorCode eMBRTUSend( UCHAR ucSlaveAddress, const UCHAR * pucFrame, USHORT usLength )
191 {
192  eMBErrorCode eStatus = MB_ENOERR;
193  USHORT usCRC16;
194 
195  ENTER_CRITICAL_SECTION( );
196 
197  /* Check if the receiver is still in idle state. If not we where to
198  * slow with processing the received frame and the master sent another
199  * frame on the network. We have to abort sending the frame.
200  */
201  if( eRcvState == STATE_RX_IDLE )
202  {
203  /* First byte before the Modbus-PDU is the slave address. */
204  pucSndBufferCur = ( UCHAR * ) pucFrame - 1;
205  usSndBufferCount = 1;
206 
207  /* Now copy the Modbus-PDU into the Modbus-Serial-Line-PDU. */
208  pucSndBufferCur[MB_SER_PDU_ADDR_OFF] = ucSlaveAddress;
209  usSndBufferCount += usLength;
210 
211  /* Calculate CRC16 checksum for Modbus-Serial-Line-PDU. */
212  usCRC16 = usMBCRC16( ( UCHAR * ) pucSndBufferCur, usSndBufferCount );
213  ucRTUBuf[usSndBufferCount++] = ( UCHAR )( usCRC16 & 0xFF );
214  ucRTUBuf[usSndBufferCount++] = ( UCHAR )( usCRC16 >> 8 );
215 
216  /* Activate the transmitter. */
217  eSndState = STATE_TX_XMIT;
218  vMBPortSerialEnable( FALSE, TRUE );
219  }
220  else
221  {
222  eStatus = MB_EIO;
223  }
224  EXIT_CRITICAL_SECTION( );
225  return eStatus;
226 }

228 BOOL xMBRTUReceiveFSM( void )
229 {
230  BOOL xTaskNeedSwitch = FALSE;
231  UCHAR ucByte;
232 
233  assert( eSndState == STATE_TX_IDLE );
234 
235  /* Always read the character. */
236  ( void )xMBPortSerialGetByte( ( CHAR * ) & ucByte );
237 
238  switch ( eRcvState )
239  {
240  /* If we have received a character in the init state we have to
241  * wait until the frame is finished.
242  */
243  case STATE_RX_INIT:
244  vMBPortTimersEnable();
245  break;
246 
247  /* In the error state we wait until all characters in the
248  * damaged frame are transmitted.
249  */
250  case STATE_RX_ERROR:
251  vMBPortTimersEnable();
252  break;
253 
254  /* In the idle state we wait for a new character. If a character
255  * is received the t1.5 and t3.5 timers are started and the
256  * receiver is in the state STATE_RX_RECEIVE.
257  */
258  case STATE_RX_IDLE:
259  ucRTUBuf[usRcvBufferPos++] = ucByte;
260  eRcvState = STATE_RX_RCV;
261 
262  /* Enable t3.5 timers. */
263  vMBPortTimersEnable();
264  break;
265 
266  /* We are currently receiving a frame. Reset the timer after
267  * every character received. If more than the maximum possible
268  * number of bytes in a modbus frame is received the frame is
269  * ignored.
270  */
271  case STATE_RX_RCV:
272  if( usRcvBufferPos < MB_SER_PDU_SIZE_MAX )
273  {
274  ucRTUBuf[usRcvBufferPos++] = ucByte;
275  }
276  else
277  {
278  eRcvState = STATE_RX_ERROR;
279  }
280  vMBPortTimersEnable();
281  break;
282  }
283  return xTaskNeedSwitch;
284 }

286 BOOL xMBRTUTransmitFSM( void )
287 {
288  BOOL xNeedPoll = FALSE;
289 
290  assert( eRcvState == STATE_RX_IDLE );
291  switch ( eSndState )
292  {
293  /* We should not get a transmitter event if the transmitter is in
294  * idle state. */
295  case STATE_TX_IDLE:
296  /* enable receiver/disable transmitter. */
297  vMBPortSerialEnable( TRUE, FALSE );
298  break;
299 
300  case STATE_TX_XMIT:
301  /* check if we are finished. */
302  if( usSndBufferCount != 0 )
303  {
304  xMBPortSerialPutByte( ( CHAR )*pucSndBufferCur );
305  pucSndBufferCur++; /* next byte in sendbuffer. */
306  usSndBufferCount--;
307  }
308  else
309  {
310  xNeedPoll = xMBPortEventPost( EV_FRAME_SENT );
311  /* Disable transmitter. This prevents another transmit buffer
312  * empty interrupt. */
313  vMBPortSerialEnable( TRUE, FALSE );
314  eSndState = STATE_TX_IDLE;
315  }
316  break;
317  }
318 
319  return xNeedPoll;
320 }

322 BOOL xMBRTUTimerT35Expired( void )
323 {
324  BOOL xNeedPoll = FALSE;
325 
326  switch ( eRcvState )
327  {
328  /* Timer t35 expired. Startup phase is finished. */
329  case STATE_RX_INIT:
330  xNeedPoll = xMBPortEventPost( EV_READY );
331  break;
332 
333  /* A frame was received and t35 expired. Notify the listener that
334  * a new frame was received. */
335  case STATE_RX_RCV:
336  xNeedPoll = xMBPortEventPost( EV_FRAME_RECEIVED );
337  break;
338 
339  /* An error occured while receiving the frame. */
340  case STATE_RX_ERROR:
341  break;
342 
343  /* Function called in an illegal state. */
344  default:
345  assert( ( eRcvState == STATE_RX_INIT ) ||
346  ( eRcvState == STATE_RX_RCV ) || ( eRcvState == STATE_RX_ERROR ) );
347  }
348 
349  vMBPortTimersDisable( );
350  eRcvState = STATE_RX_IDLE;
351 
352  return xNeedPoll;
353 }
