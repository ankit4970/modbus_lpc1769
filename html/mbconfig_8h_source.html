38 /*! \defgroup modbus_cfg Modbus Configuration
39  *
40  * Most modules in the protocol stack are completly optional and can be
41  * excluded. This is specially important if target resources are very small
42  * and program memory space should be saved.<br>
43  *
44  * All of these settings are available in the file <code>mbconfig.h</code>
45  */
46 /*! \addtogroup modbus_cfg
47  * @{
48  */
49 /*! \brief If Modbus ASCII support is enabled. */
50 #define MB_ASCII_ENABLED ( 0 )
51 /*! \brief If Modbus RTU support is enabled. */
52 #define MB_RTU_ENABLED ( 1 )
53 /*! \brief If Modbus TCP support is enabled. */
54 #define MB_TCP_ENABLED ( 0 )
55 /*! \brief The character timeout value for Modbus ASCII.
56  *
57  * The character timeout value is not fixed for Modbus ASCII and is therefore
58  * a configuration option. It should be set to the maximum expected delay
59  * time of the network.
60  */
61 #define MB_ASCII_TIMEOUT_SEC ( 0 )
62 /*! \brief Maximum number of Modbus functions codes the protocol stack
63  * should support.
64  *
65  * The maximum number of supported Modbus functions must be greater than
66  * the sum of all enabled functions in this file and custom function
67  * handlers. If set to small adding more functions will fail.
68  */
69 #define MB_FUNC_HANDLERS_MAX ( 16 )
70 /*! \brief Number of bytes which should be allocated for the <em>Report Slave ID
71  * </em>command.
72  *
73  * This number limits the maximum size of the additional segment in the
74  * report slave id function. See eMBSetSlaveID( ) for more information on
75  * how to set this value. It is only used if MB_FUNC_OTHER_REP_SLAVEID_ENABLED
76  * is set to <code>1</code>.
77  */
78 #define MB_FUNC_OTHER_REP_SLAVEID_BUF ( 32 )
79 /*! \brief If the <em>Report Slave ID</em> function should be enabled. */
80 #define MB_FUNC_OTHER_REP_SLAVEID_ENABLED ( 1 )
81 /*! \brief If the <em>Read Input Registers</em> function should be enabled. */
82 #define MB_FUNC_READ_INPUT_ENABLED ( 1 )
83 /*! \brief If the <em>Read Holding Registers</em> function should be enabled. */
84 #define MB_FUNC_READ_HOLDING_ENABLED ( 1 )
85 /*! \brief If the <em>Write Single Register</em> function should be enabled. */
86 #define MB_FUNC_WRITE_HOLDING_ENABLED ( 1 )
87 /*! \brief If the <em>Write Multiple registers</em> function should be enabled. */
88 #define MB_FUNC_WRITE_MULTIPLE_HOLDING_ENABLED ( 1 )
89 /*! \brief If the <em>Read Coils</em> function should be enabled. */
90 #define MB_FUNC_READ_COILS_ENABLED ( 1 )
91 /*! \brief If the <em>Write Coils</em> function should be enabled. */
92 #define MB_FUNC_WRITE_COIL_ENABLED ( 1 )
93 /*! \brief If the <em>Write Multiple Coils</em> function should be enabled. */
94 #define MB_FUNC_WRITE_MULTIPLE_COILS_ENABLED ( 1 )
95 /*! \brief If the <em>Read Discrete Inputs</em> function should be enabled. */
96 #define MB_FUNC_READ_DISCRETE_INPUTS_ENABLED ( 1 )
97 /*! \brief If the <em>Read/Write Multiple Registers</em> function should be enabled. */
98 #define MB_FUNC_READWRITE_HOLDING_ENABLED ( 1 )
99 /*! @} */
