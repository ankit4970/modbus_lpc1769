52 void delay(uint32_t delayInMs)
53 {
54  LPC_TIM0->TCR = 0x02; /* reset timer */
55  LPC_TIM0->PR = 0x00; /* set prescaler to zero */
56  LPC_TIM0->MR0 = delayInMs * (9000000 / 1000-1);
57  LPC_TIM0->IR = 0xff; /* reset all interrrupts */
58  LPC_TIM0->MCR = 0x04; /* stop timer on match */
59  LPC_TIM0->TCR = 0x01; /* start timer */
60 
61  /* wait until delay time has elapsed */
62  while (LPC_TIM0->TCR & 0x01);
63 
64  return;
65 }

67 unsigned int sys_get_cpu_clock()
68 {
69  unsigned clock = 0;
70 
71  /* Determine clock frequency according to clock register values */
72  if (((LPC_SC->PLL0STAT >> 24) & 3) == 3)
73  { /* If PLL0 enabled and connected */
74  switch (LPC_SC->CLKSRCSEL & 0x03)
75  {
76  case 0: /* Int. RC oscillator => PLL0 */
77  case 3: /* Reserved, default to Int. RC */
78  clock = (INTERNAL_CLOCK
79  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
80  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
81  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
82  break;
83 
84  case 1: /* Main oscillator => PLL0 */
85  clock = (EXTERNAL_CLOCK
86  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
87  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
88  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
89  break;
90 
91  case 2: /* RTC oscillator => PLL0 */
92  clock = (RTC_CLOCK
93  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
94  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
95  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
96  break;
97  }
98  }
99  else
100  {
101  switch (LPC_SC->CLKSRCSEL & 0x03)
102  {
103  case 0: /* Int. RC oscillator => PLL0 */
104  case 3: /* Reserved, default to Int. RC */
105  clock = INTERNAL_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
106  break;
107  case 1: /* Main oscillator => PLL0 */
108  clock = EXTERNAL_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
109  break;
110  case 2: /* RTC oscillator => PLL0 */
111  clock = RTC_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
112  break;
113  }
114  }
115 
116  return clock;
117 }

121 void
122 xMBUtilSetBits( UCHAR * ucByteBuf, USHORT usBitOffset, UCHAR ucNBits,
123  UCHAR ucValue )
124 {
125  USHORT usWordBuf;
126  USHORT usMask;
127  USHORT usByteOffset;
128  USHORT usNPreBits;
129  USHORT usValue = ucValue;
130 
131  assert( ucNBits <= 8 );
132  assert( ( size_t )BITS_UCHAR == sizeof( UCHAR ) * 8 );
133 
134  /* Calculate byte offset for first byte containing the bit values starting
135  * at usBitOffset. */
136  usByteOffset = ( USHORT )( ( usBitOffset ) / BITS_UCHAR );
137 
138  /* How many bits precede our bits to set. */
139  usNPreBits = ( USHORT )( usBitOffset - usByteOffset * BITS_UCHAR );
140 
141  /* Move bit field into position over bits to set */
142  usValue <<= usNPreBits;
143 
144  /* Prepare a mask for setting the new bits. */
145  usMask = ( USHORT )( ( 1 << ( USHORT ) ucNBits ) - 1 );
146  usMask <<= usBitOffset - usByteOffset * BITS_UCHAR;
147 
148  /* copy bits into temporary storage. */
149  usWordBuf = ucByteBuf[usByteOffset];
150  usWordBuf |= ucByteBuf[usByteOffset + 1] << BITS_UCHAR;
151 
152  /* Zero out bit field bits and then or value bits into them. */
153  usWordBuf = ( USHORT )( ( usWordBuf & ( ~usMask ) ) | usValue );
154 
155  /* move bits back into storage */
156  ucByteBuf[usByteOffset] = ( UCHAR )( usWordBuf & 0xFF );
157  ucByteBuf[usByteOffset + 1] = ( UCHAR )( usWordBuf >> BITS_UCHAR );
158 }

160 UCHAR
161 xMBUtilGetBits( UCHAR * ucByteBuf, USHORT usBitOffset, UCHAR ucNBits )
162 {
163  USHORT usWordBuf;
164  USHORT usMask;
165  USHORT usByteOffset;
166  USHORT usNPreBits;
167 
168  /* Calculate byte offset for first byte containing the bit values starting
169  * at usBitOffset. */
170  usByteOffset = ( USHORT )( ( usBitOffset ) / BITS_UCHAR );
171 
172  /* How many bits precede our bits to set. */
173  usNPreBits = ( USHORT )( usBitOffset - usByteOffset * BITS_UCHAR );
174 
175  /* Prepare a mask for setting the new bits. */
176  usMask = ( USHORT )( ( 1 << ( USHORT ) ucNBits ) - 1 );
177 
178  /* copy bits into temporary storage. */
179  usWordBuf = ucByteBuf[usByteOffset];
180  usWordBuf |= ucByteBuf[usByteOffset + 1] << BITS_UCHAR;
181 
182  /* throw away unneeded bits. */
183  usWordBuf >>= usNPreBits;
184 
185  /* mask away bits above the requested bitfield. */
186  usWordBuf &= usMask;
187 
188  return ( UCHAR ) usWordBuf;
189 }
