226 //*****************************************************************************
227 // Functions to carry out the initialization of RW and BSS data sections. These
228 // are written as separate functions rather than being inlined within the
229 // ResetISR() function in order to cope with MCUs with multiple banks of
230 // memory.
231 //*****************************************************************************
232 __attribute__ ((section(".after_vectors")))
233 void data_init(unsigned int romstart, unsigned int start, unsigned int len) {
234  unsigned int *pulDest = (unsigned int*) start;
235  unsigned int *pulSrc = (unsigned int*) romstart;
236  unsigned int loop;
237  for (loop = 0; loop < len; loop = loop + 4)
238  *pulDest++ = *pulSrc++;
239 }
240 
241 __attribute__ ((section(".after_vectors")))
242 void bss_init(unsigned int start, unsigned int len) {
243  unsigned int *pulDest = (unsigned int*) start;
244  unsigned int loop;
245  for (loop = 0; loop < len; loop = loop + 4)
246  *pulDest++ = 0;
247 }

249 //*****************************************************************************
250 // The following symbols are constructs generated by the linker, indicating
251 // the location of various points in the "Global Section Table". This table is
252 // created by the linker via the Code Red managed linker script mechanism. It
253 // contains the load address, execution address and length of each RW data
254 // section and the execution and length of each BSS (zero initialized) section.
255 //*****************************************************************************
256 extern unsigned int __data_section_table;
257 extern unsigned int __data_section_table_end;
258 extern unsigned int __bss_section_table;
259 extern unsigned int __bss_section_table_end;
260 
261 //*****************************************************************************
262 // Reset entry point for your code.
263 // Sets up a simple runtime environment and initializes the C/C++
264 // library.
265 //*****************************************************************************
266 __attribute__ ((section(".after_vectors")))
267 void
268 ResetISR(void) {
269 
270  //
271  // Copy the data sections from flash to SRAM.
272  //
273  unsigned int LoadAddr, ExeAddr, SectionLen;
274  unsigned int *SectionTableAddr;
275 
276  // Load base address of Global Section Table
277  SectionTableAddr = &__data_section_table;
278 
279  // Copy the data sections from flash to SRAM.
280  while (SectionTableAddr < &__data_section_table_end) {
281  LoadAddr = *SectionTableAddr++;
282  ExeAddr = *SectionTableAddr++;
283  SectionLen = *SectionTableAddr++;
284  data_init(LoadAddr, ExeAddr, SectionLen);
285  }
286  // At this point, SectionTableAddr = &__bss_section_table;
287  // Zero fill the bss segment
288  while (SectionTableAddr < &__bss_section_table_end) {
289  ExeAddr = *SectionTableAddr++;
290  SectionLen = *SectionTableAddr++;
291  bss_init(ExeAddr, SectionLen);
292  }
293 
294 #if defined (__USE_CMSIS) || defined (__USE_LPCOPEN)
295  SystemInit();
296 #endif
297 
298 #if defined (__cplusplus)
299  //
300  // Call C++ library initialisation
301  //
302  __libc_init_array();
303 #endif
304 
305 #if defined (__REDLIB__)
306  // Call the Redlib library, which in turn calls main()
307  __main() ;
308 #else
309  main();
310 #endif
311 
312  //
313  // main() shouldn't return, but if it does, we'll just enter an infinite loop
314  //
315  while (1) {
316  ;
317  }
318 }
