120 BOOL xMBPortSerialInit( UCHAR ucPORT, ULONG ulBaudRate, UCHAR ucDataBits, eMBParity eParity )
121 {
122  //simISR.attach_us(&prvvUARTISR,1000); // Cam - attach prvvUARTISR to a 1mS ticker to simulate serial interrupt behaviour
123  // 1mS is just short of a character time at 9600 bps, so quick enough to pick
124  // up status on a character by character basis.
125 
126  //const uint32_t Baud_Rate = 9600;
127 
128  // Step 1: UART2 power/clock control bit(24) of PCONP
129  LPC_SC->PCONP &=~ (1<<24);
130  LPC_SC->PCONP |= (1<<24);
131 
132  // Step 2: PCLKSEL1 bit 16 and 17 of Peripheral Clock Selection register1 for Uart2, controls the rate of the clock signal
133  // that will be supplied to the corresponding peripheral Function peripheral clock selection
134  // 00 CCLK/4
135  // 01 CCLK
136  // 10 CCLK/4
137  // 11 CLK/4
138  LPC_SC->PCLKSEL1 &=~(3<<16);
139  LPC_SC->PCLKSEL1 |= (1<<16);
140 
141  // Step 3:// table 280
142  LPC_UART2->LCR &=~ (3<<0 | 1<<7);
143 
144  // LCR bit o and 1, Word length select
145  LPC_UART2->LCR |= (3<<0); //00 5-bit character length
146  //01 6-bit character length
147  //10 7-bit character length
148  //11 8-bit character length
149 
150  //Enable, Divisor latch access bit(DLAB)
151  LPC_UART2->LCR |= (1<<7);
152  LPC_UART2->DLM = 0x00;
153  //LPC_UART3->DLL =0x38;
154  LPC_UART2->DLL =( sys_get_cpu_clock() / (16 * ulBaudRate));
155 
156  // Step 4: FIFO not use:
157  LPC_PINCON->PINSEL0 &=~ (3<<20 | 3<<22);
158  LPC_PINCON->PINSEL0 |= (1<<20 | 1<<22); //TXD3 | RXD3
159  LPC_PINCON->PINMODE4 &=~ (3<<20 | 3<<22);
160  LPC_PINCON->PINMODE4 |= (3<<22); // 11 pull down
161 
162  // Step 6: Disable, Divisor latch access bit(DLAB)
163  LPC_UART2->LCR &=~ (1<<7);
164  NVIC_EnableIRQ(UART2_IRQn);
165 
166  //enable receive data available interrupt for uart2
167  LPC_UART2->IER &=~ (1<<0);
168  LPC_UART2->IER |= (1<<0);
169 
170  return TRUE;
171 }
