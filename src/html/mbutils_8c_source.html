55  delay
56  -----------------------------------------------------------------------------------------------------------------------
57 * Event Handler for GPI module
58 *
59 * @date DEC/02/2013
60 * @author FW_DEV_2
61 * @pre None
62 * @return None
63 ************************************************************************************************************************
64 */
65 void delay(uint32_t delayInMs)
66 {
67  LPC_TIM0->TCR = 0x02; /* reset timer */
68  LPC_TIM0->PR = 0x00; /* set prescaler to zero */
69  LPC_TIM0->MR0 = delayInMs * (9000000 / 1000-1);
70  LPC_TIM0->IR = 0xff; /* reset all interrrupts */
71  LPC_TIM0->MCR = 0x04; /* stop timer on match */
72  LPC_TIM0->TCR = 0x01; /* start timer */
73 
74  /* wait until delay time has elapsed */
75  while (LPC_TIM0->TCR & 0x01);
76 
77  return;
78 }

82  sys_get_cpu_clock
83  -----------------------------------------------------------------------------------------------------------------------
84 * Event Handler for GPI module
85 *
86 * @date DEC/02/2013
87 * @author FW_DEV_2
88 * @pre None
89 * @return None
90 ************************************************************************************************************************
91 */
92 unsigned int sys_get_cpu_clock()
93 {
94  unsigned clock = 0;
95 
96  /* Determine clock frequency according to clock register values */
97  if (((LPC_SC->PLL0STAT >> 24) & 3) == 3)
98  { /* If PLL0 enabled and connected */
99  switch (LPC_SC->CLKSRCSEL & 0x03)
100  {
101  case 0: /* Int. RC oscillator => PLL0 */
102  case 3: /* Reserved, default to Int. RC */
103  clock = (INTERNAL_CLOCK
104  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
105  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
106  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
107  break;
108 
109  case 1: /* Main oscillator => PLL0 */
110  clock = (EXTERNAL_CLOCK
111  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
112  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
113  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
114  break;
115 
116  case 2: /* RTC oscillator => PLL0 */
117  clock = (RTC_CLOCK
118  * ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1)))
119  / (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1)
120  / ((LPC_SC->CCLKCFG & 0xFF) + 1));
121  break;
122  }
123  }
124  else
125  {
126  switch (LPC_SC->CLKSRCSEL & 0x03)
127  {
128  case 0: /* Int. RC oscillator => PLL0 */
129  case 3: /* Reserved, default to Int. RC */
130  clock = INTERNAL_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
131  break;
132  case 1: /* Main oscillator => PLL0 */
133  clock = EXTERNAL_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
134  break;
135  case 2: /* RTC oscillator => PLL0 */
136  clock = RTC_CLOCK / ((LPC_SC->CCLKCFG & 0xFF) + 1);
137  break;
138  }
139  }
140 
141  return clock;
142 }

158 void xMBUtilSetBits( UCHAR * ucByteBuf, USHORT usBitOffset, UCHAR ucNBits,
159  UCHAR ucValue )
160 {
161  USHORT usWordBuf;
162  USHORT usMask;
163  USHORT usByteOffset;
164  USHORT usNPreBits;
165  USHORT usValue = ucValue;
166 
167  assert( ucNBits <= 8 );
168  assert( ( size_t )BITS_UCHAR == sizeof( UCHAR ) * 8 );
169 
170  /* Calculate byte offset for first byte containing the bit values starting
171  * at usBitOffset. */
172  usByteOffset = ( USHORT )( ( usBitOffset ) / BITS_UCHAR );
173 
174  /* How many bits precede our bits to set. */
175  usNPreBits = ( USHORT )( usBitOffset - usByteOffset * BITS_UCHAR );
176 
177  /* Move bit field into position over bits to set */
178  usValue <<= usNPreBits;
179 
180  /* Prepare a mask for setting the new bits. */
181  usMask = ( USHORT )( ( 1 << ( USHORT ) ucNBits ) - 1 );
182  usMask <<= usBitOffset - usByteOffset * BITS_UCHAR;
183 
184  /* copy bits into temporary storage. */
185  usWordBuf = ucByteBuf[usByteOffset];
186  usWordBuf |= ucByteBuf[usByteOffset + 1] << BITS_UCHAR;
187 
188  /* Zero out bit field bits and then or value bits into them. */
189  usWordBuf = ( USHORT )( ( usWordBuf & ( ~usMask ) ) | usValue );
190 
191  /* move bits back into storage */
192  ucByteBuf[usByteOffset] = ( UCHAR )( usWordBuf & 0xFF );
193  ucByteBuf[usByteOffset + 1] = ( UCHAR )( usWordBuf >> BITS_UCHAR );
194 }

218 
219  /* How many bits precede our bits to set. */
220  usNPreBits = ( USHORT )( usBitOffset - usByteOffset * BITS_UCHAR );
221

223  usMask = ( USHORT )( ( 1 << ( USHORT ) ucNBits ) - 1 );
224 
225  /* copy bits into temporary storage. */
226  usWordBuf = ucByteBuf[usByteOffset];
227  usWordBuf |= ucByteBuf[usByteOffset + 1] << BITS_UCHAR;
228 
229  /* throw away unneeded bits. */
230  usWordBuf >>= usNPreBits;
231 
232  /* mask away bits above the requested bitfield. */
233  usWordBuf &= usMask;
234 
235  return ( UCHAR ) usWordBuf;
