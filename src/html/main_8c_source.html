88 eMBErrorCode eMBRegInputCB( UCHAR * pucRegBuffer, USHORT usAddress, USHORT usNRegs )
89 {
90  eMBErrorCode eStatus = MB_ENOERR;
91  int iRegIndex;
92 
93  if( ( usAddress >= REG_INPUT_START ) && ( usAddress + usNRegs <= REG_INPUT_START + REG_INPUT_NREGS ) )
94  {
95  iRegIndex = ( int )( usAddress - usRegInputStart );
96  while( usNRegs > 0 )
97  {
98  *pucRegBuffer++ =
99  ( unsigned char )( usRegInputBuf[iRegIndex] >> 8 );
100  *pucRegBuffer++ =
101  ( unsigned char )( usRegInputBuf[iRegIndex] & 0xFF );
102  iRegIndex++;
103  usNRegs--;
104  }
105  }
106  else
107  {
108  eStatus = MB_ENOREG;
109  }
110 
111  return eStatus;
112 }
